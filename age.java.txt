import java.util.Scanner;

public class age
{
    public static void main(String[] args)
    {
        int age;
        Scanner sc = new Scanner(System.in);
        age = sc.nextInt();
        if (age >= 18)
        {
            System.out.println("You are eligible for vote.");
        }
        else
        {
            System.out.println("You are not eligible for vote.");
        }
    }
}
